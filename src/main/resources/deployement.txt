apiVersion: apps/v1
kind: Deployment
metadata:
  name: docker-kubernetes-deployment
spec:
  replicas: 3
  selector:
    matchLabels:
      app: docker-kubernetes
  template:
    metadata:
      labels:
        app: docker-kubernetes
    spec:
      containers:
        - name: spring-app
          image: docker-kubernetes:latest
          imagePullPolicy: Never  # CRITICAL: Tells K8s to look only inside Minikube
          ports:
            - containerPort: 8080
---
apiVersion: v1
kind: Service
metadata:
  name: docker-kubernetes-service
spec:
  type: NodePort
  selector:
    app: docker-kubernetes
  ports:
    - port: 80
      targetPort: 8080
      nodePort: 30001
